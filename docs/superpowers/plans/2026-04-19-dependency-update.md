```dart
/// Abstract base Notifier for list operations.
/// Concrete subclasses per entity provide [service] and [config].
abstract class DataListNotifier<T> extends Notifier<DataListState<T>> {
  ApiConfig<T> get config;

  late final ApiService<T> service = ApiServiceFactory.createCRUD(config);

  @override
  DataListState<T> build() {
    loadData();
    return DataListState<T>();
  }

  Future<void> loadData() async {
    state = state.copyWith(isLoading: true, error: null);

    final result = await service.getAll();
    result.fold(
      (failure) =>
          state = state.copyWith(isLoading: false, error: failure.message),
      (items) =>
          state = state.copyWith(isLoading: false, items: items, error: null),
    );
  }

  Future<void> refresh() async {
    state = state.copyWith(isRefreshing: true);

    final result = await service.getAll();
    result.fold(
      (failure) =>
          state = state.copyWith(isRefreshing: false, error: failure.message),
      (items) => state = state.copyWith(
        isRefreshing: false,
        items: items,
        error: null,
      ),
    );
  }

  Future<void> create(Map<String, dynamic> data) async {
    if (config.toJson == null) return;

    final item = config.fromJson(data);
    final result = await service.create(item);
    result.fold(
      (failure) => state = state.copyWith(error: failure.message),
      (newItem) {
        final updatedItems = <T>[...state.items, newItem];
        state = state.copyWith(items: updatedItems, error: null);
      },
    );
  }

  Future<void> update(T item, Map<String, dynamic> data) async {
    if (config.toJson == null || config.getId == null) return;

    final existingJson = config.toJson!(item);
    final mergedJson = {...existingJson, ...data};
    final mergedItem = config.fromJson(mergedJson);
    final result = await service.update(mergedItem);
    result.fold(
      (failure) => state = state.copyWith(error: failure.message),
      (updatedItem) {
        final updatedItems = state.items
            .map((i) => i == item ? updatedItem : i)
            .toList();
        state = state.copyWith(items: updatedItems, error: null);
      },
    );
  }

  Future<void> delete(T item) async {
    if (config.getId == null) return;

    final result = await service.delete(item);
    result.fold(
      (failure) => state = state.copyWith(error: failure.message),
      (_) {
        final updatedItems = state.items.where((i) => i != item).toList();
        state = state.copyWith(items: updatedItems, error: null);
      },
    );
  }

  void clearError() {
    state = state.copyWith(error: null);
  }
}
```

```dart
/// Abstract base Notifier for single item operations.
/// Note: Unlike the old DataItemNotifier, this auto-loads data in build().
/// The old version required manual loadData() calls. This is an intentional
/// behavioral change to match DataListNotifier's auto-load pattern.
abstract class DataItemNotifier<T> extends Notifier<DataItemState<T>> {
  Future<Either<Failure, T>> fetchItem();

  @override
  DataItemState<T> build() {
    loadData();
    return DataItemState<T>();
  }

  Future<void> loadData() async {
    state = state.copyWith(isLoading: true, error: null);

    final result = await fetchItem();
    result.fold(
      (failure) =>
          state = state.copyWith(isLoading: false, error: failure.message),
      (item) =>
          state = state.copyWith(isLoading: false, item: item, error: null),
    );
  }

  void clearError() {
    state = state.copyWith(error: null);
  }
}

  PostsNotifier get postsNotifier => read(ApiProviders.postsProvider.notifier);
}
```

Note on constructor params: `UserByIdNotifier` and `PostByIdNotifier` take constructor params (`userId`/`postId`). `NotifierProvider` accepts a `NotifierT Function()` factory, so `() => UserByIdNotifier(id)` is valid — Riverpod 3 does not require a no-arg constructor, it requires a factory function. If this approach causes compilation issues at runtime, the fallback is to replace these with `FutureProvider.family`:

```dart
static final userByIdProvider = FutureProvider.family<User, int>((ref, id) async {
  final service = ApiServiceFactory.createReadOnly(ApiConfig<User>(
    endpoint: '/users',
    fromJson: User.fromJson,
  ));
  final result = await service.getById(id);
  return result.fold((f) => throw Exception(f.message), (user) => user);
});
```

`ref.invalidate()` is the Riverpod 3 idiomatic way to force a provider to rebuild without needing the return value.
